MULTIPLICATION OF FLOAT


#include <stdio.h>
int main()
{
double a, b, product;
printf("Enter two numbers: ");
scanf("%lf %lf", &a, &b);
// Calculating product
product = a * b;
// Result up to 2 decimal point is displayed using %.2lf
printf("Product = %.2lf", product);
return 0;
}

OUTPUT


Enter two numbers: 2.4
1.12
Product = 2.69
