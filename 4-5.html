ASSIGNMENT OPERATOR


#include <stdio.h>
int main()
{
int a = 7;
int Total = 21;
printf(" Value of the Total = %d \n", Total += a );
printf(" Value of the Total = %d \n", Total -= a );
printf(" Value of the Total = %d \n", Total *= a );
printf(" Value of the Total = %d \n", Total /= a );
printf(" Value of the Total = %d \n", Total %= a );
return 0;
}
