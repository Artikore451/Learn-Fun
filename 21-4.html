ACCESSING USNIO MEMBERS


#include <stdio.h>
union Job
{
float salary;
int workerNo;
} j;

int main()
{
j.salary = 12.3;

// when j.workerNo is assigned a value,
// j.salary will no longer hold 12.3
j.workerNo = 100;

printf("Salary = %.1f\n", j.salary);
printf("Number of workers = %d", j.workerNo);
return 0;
}

OUTPUT


Salary = 0.0
Number of workers = 100
