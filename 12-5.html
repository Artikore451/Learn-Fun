DISPLAY 3*3 MATRIX OF GIVEN ELEMENTS


#include <stdio.h>
void matrix( int m[][3] )
{
int i,j;
printf("Matrix is :\n");
for ( i=0;i<3;i++ )
{
for ( j=0;j<3;j++)
{
printf("%d",m[i][j]);
}
printf("\n");
}
}

int main()
{
int m[3][3] = {1,2,3,4,5,6,7,8,9};
matrix(m);
return 0;
}

OUTPUT


Matrix is :
1 2 3
4 5 6
7 8 9
