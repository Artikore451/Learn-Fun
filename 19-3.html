PROGRAM FOR REALLOC() AND FREE() FUNCTIONS


#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main()
{
char *mem_allocation;
/* memory is allocated dynamically */
mem_allocation = malloc( 20 * sizeof(char) );
if( mem_allocation == NULL )
{
printf("Couldn't able to allocate requested memory\n");
}
else
{
strcpy( mem_allocation,"artikore451.com");
}
printf("Dynamically allocated memory content : " \ "%s\n", mem_allocation );
mem_allocation=realloc(mem_allocation,100*sizeof(char));
if( mem_allocation == NULL )
{
printf("Couldn't able to allocate requested memory\n"); }
else
{
strcpy( mem_allocation,"space is extended upto " \ "100 characters");
}

printf("R
esized memory : %s\n", mem_allocation );
free(mem_allocation);
}
