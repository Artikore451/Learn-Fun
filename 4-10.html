RELATIONAL OPERATOR


#include <stdio.h>
int main()
{
int a = 9;
int b = 4;
printf(" a > b: %d \n", a > b);
printf("a >= b: %d \n", a >= b);
printf("a <= b: %d \n", a <= b);
printf("a < b: %d \n", a < b);
printf("a == b: %d \n", a == b);
printf("a != b: %d \n", a != b);
}

OUTPUT


a > b:1
a >= b:1
a <= b:0
a < b:0
a == b:0
a != b: 1
